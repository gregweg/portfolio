Recursive Python function to build a quadtree from a grid input

def constructQuadTree(self, grid: List[List[int]]) -> 'Node':
        current_grid_equal, first_val = self.allGridValuesEqual(grid)

        if current_grid_equal:
            # If the current grid has the same value
            # Set isLeaf: True and set val to the value of the grid
            # set four children to Null and stop
            return Node(first_val, True, None, None, None, None)
        else:
            # If current grid has diff values, set isLeaf: False
            # set val to any value
            # divide current grid into 4 sub-grids and recurse
            h, w = len(grid), len(grid[0])
            mid_h, mid_w = h // 2, w // 2
            top_left =  [row[:mid_w] for row in grid[:mid_h]]
            top_right = [row[mid_w:] for row in grid[:mid_h]]
            bot_left =  [row[:mid_w] for row in grid[mid_h:]]
            bot_right = [row[mid_w:] for row in grid[mid_h:]]
            topLeft = self.construct(top_left)
            topRight = self.construct(top_right)
            bottomLeft = self.construct(bot_left)
            bottomRight = self.construct(bot_right)
            return Node(first_val, False, topLeft, topRight, bottomLeft, bottomRight)

    def allGridValuesEqual(self, grid: List[List[int]]) -> bool:
        if not grid:
            return True
        first_val = grid[0][0]

        return (all(all(cell == first_val for cell in row) for row in grid), first_val)

Python code implementing efficient sliding window approach with min/max tracking queues for longest subarray problems

def longestSubarray(self, nums: List[int], limit: int) -> int:
        max_queue = deque()
        min_queue = deque()

        left = 0
        longest = 0

        for right, num in enumerate(nums):
            while max_queue and num > max_queue[-1]:
                max_queue.pop()
            max_queue.append(num)

            while min_queue and num < min_queue[-1]:
                min_queue.pop()
            min_queue.append(num)

            while max_queue[0] - min_queue[0] > limit:
                if nums[left] == max_queue[0]:
                    max_queue.popleft()

                if nums[left] == min_queue[0]:
                    min_queue.popleft()

                left += 1
            longest = max(longest, right - left + 1)

        return longest

Python class modeling dynamic group formation and union over time, optimized with binary masking

from dataclasses import dataclass

@dataclass
class FriendGroup:
    members: set[int]
    mask: int

    def earliestAcq(self, logs: List[List[int]], n: int) -> int:
        logs.sort()
        friend_groups: List[FriendGroup] = []
        full_n = (1 << n) - 1
        friends_seen = set()
        for ts, p_one, p_two in logs:
            p_one_mask = 1 << p_one
            p_two_mask = 1 << p_two
            p_one_is_popular = p_one in friends_seen
            p_two_is_popular = p_two in friends_seen
            friends_seen.add(p_one)
            friends_seen.add(p_two)

            if p_one_is_popular or p_two_is_popular:
                a, b = None, None
                for group_idx, friend_group in enumerate(friend_groups):
                    if p_one in friend_group.members:
                        a = group_idx
                    if p_two in friend_group.members:
                        b = group_idx

                if a is not None and b is None:
                    friend_groups[a].members.add(p_two)
                    friend_groups[a].mask |= p_two_mask
                elif a is None and b is not None:
                    friend_groups[b].members.add(p_one)
                    friend_groups[b].mask |= p_one_mask
                elif a is not None and b is not None and a != b:
                    friend_groups[a].members.update(friend_groups[b].members)
                    friend_groups[a].mask |= friend_groups[b].mask
                    del friend_groups[b]

                for group in friend_groups:
                    if group.mask == full_n:
                        return ts
            else:
                friends_mask = p_one_mask | p_two_mask
                friend_groups.append(
                    FriendGroup(
                        members={p_one, p_two},
                        mask=friends_mask,
                    )
                )
                if friends_mask == full_n:
                    return ts
        return -1

Implementation of a data structure that maintains uniqueness efficiently within an array

class FirstUnique:
    def __init__(self, nums: List[int]):
        self.counts = Counter(nums)
        self.queue = deque(nums)

    def showFirstUnique(self) -> int:
        while self.queue and self.counts[self.queue[0]] > 1:
            self.queue.popleft()
        return self.queue[0] if self.queue else -1

    def add(self, value: int) -> None:
        self.counts[value] += 1
        self.queue.append(value)

Exploring how to use bisect_right for finding the rightmost element’s index

def findPeakElement(self, nums: List[int]) -> int:
        left, right = 0, len(nums)-1

        while left < right:
            mid = (left + right) // 2

            if nums[mid] < nums[mid+1]:
                left = mid+1
            else:
                right = mid
        return left
