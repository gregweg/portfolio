Efficient Data Structures for Key Problems

Today's notes focus on optimizing data structures to solve common problems related to caching and efficient data retrieval. The theme that ties these together is the need for fast lookups, updates, and evictions in scenarios where memory usage needs to be tightly controlled but performance must remain high.

LRU Cache

LRU Cache
O(1) get
O(1) put

HashMap + Doubly Linked List
- fast lookup
- fast recency updates
- fast eviction

class Node:
    def __init__(self, key, value):
        self.key = key
        self.value = value
        self.prev = None
        self.next = None

class LRUCache:
    def __init__(self, capacity: int):
        self.cache = {}
        self.capacity = capacity

        self.head = Node(0, 0)
        self.tail = Node(0, 0)

        self.head.next = self.tail
        self.tail.prev = self.head


    def _remove(self, node):
        prev_node = node.prev
        next_node = node.next

        prev_node.next = next_node
        next_node.prev = prev_node

    def _insert_front(self, node):
        first_real_node = self.head.next

        self.head.next = node
        node.prev = self.head

        node.next = first_real_node
        first_real_node.prev = node

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1

        node = self.cache[key]
        self._remove(node)
        self._insert_front(node)

        return node.value        

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._remove(node)
        else:
            node = Node(key, value)
            self.cache[key] = node

        self._insert_front(node)

        if len(self.cache) > self.capacity:
            lru_node = self.tail.prev
            self._remove(lru_node)
            del self.cache[lru_node.key]

LFU Cache (Least Frequently Used)

O(1) get
O(1) put

HashMap + Frequency Buckets + Doubly Linked Lists

  • fast lookup
  • fast frequency updates
  • fast least-frequently-used eviction

Core Structures:

  • key_to_node
  • freq_to_list
  • min_freq tracker

Need BOTH:

  • frequency ordering
  • recency ordering
from collections import defaultdict, OrderedDict

class LFUCache:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.min_freq = 0

        self.key_to_val_freq = {}  # key -> (value, freq)
        self.freq_to_keys = defaultdict(OrderedDict)  # freq -> keys in recency order

    def _increase_freq(self, key: int) -> None:
        value, freq = self.key_to_val_freq[key]

        del self.freq_to_keys[freq][key]

        if not self.freq_to_keys[freq] and freq == self.min_freq:
            self.min_freq += 1

        new_freq = freq + 1
        self.freq_to_keys[new_freq][key] = None
        self.key_to_val_freq[key] = (value, new_freq)

    def get(self, key: int) -> int:
        if key not in self.key_to_val_freq:
            return -1

        self._increase_freq(key)
        return self.key_to_val_freq[key][0]

    def put(self, key: int, value: int) -> None:
        if self.capacity == 0:
            return

        if key in self.key_to_val_freq:
            _, freq = self.key_to_val_freq[key]
            self.key_to_val_freq[key] = (value, freq)
            self._increase_freq(key)
            return

        if len(self.key_to_val_freq) >= self.capacity:
            lfu_key, _ = self.freq_to_keys[self.min_freq].popitem(last=False)
            del self.key_to_val_freq[lfu_key]

        self.key_to_val_freq[key] = (value, 1)
        self.freq_to_keys[1][key] = None
        self.min_freq = 1

Median From Data Stream

insert -> O(log n)
findMedian -> O(1)

Max Heap + Min Heap

  • maintain lower half
  • maintain upper half
  • efficient median lookup

Purpose:
Efficiently maintain median while continuously inserting values.

Core Idea:
Split values into:

  • lower half
  • upper half

Data Structures:

  • max heap -> smaller half
  • min heap -> larger half
def __init__(self):
    self.small = []  # max heap
    self.large = []  # min heap

def addNum(self, num: int) -> None:
    heapq.heappush(self.small, -num)

    heapq.heappush(self.large, -heapq.heappop(self.small))

    if len(self.large) > len(self.small):
        heapq.heappush(self.small, -heapq.heappop(self.large))

def findMedian(self) -> float:
    if len(self.small) > len(self.large):
        return -self.small[0]

    return (-self.small[0] + self.large[0]) / 2

Randomized Set

Runtime:

insert: O(1) average
remove: O(1) average

import random


class RandomizedSet:
    def __init__(self):
        self.indices = {}
        self.number_list = []

    # Runtime: O(1) average
    def insert(self, val: int) -> bool:
        if val in self.indices:
            return False
        self.indices[val] = len(self.number_list)
        self.number_list.append(val)
        return True

    # Runtime: O(1) average
    def remove(self, val: int) -> bool:
        if val not in self.indices:
            return False
        val_idx = self.indices[val]
        last_val = self.number_list[-1]

        self.number_list[val_idx] = last_val
        self.indices[last_val] = val_idx

        self.number_list.pop()
        del self.indices[val]

        return True

    # Runtime: O(1)
    def getRandom(self) -> int:
        return random.choice(self.number_list)

Min Stack

Min Stack

O(1) push
O(1) pop
O(1) top
O(1) getMin

Stack + Auxiliary Min Stack

  • fast push/pop
  • fast current minimum lookup

Purpose:
Support normal stack operations while retrieving minimum value in O(1).

Core Idea:
Maintain TWO stacks:

  • regular stack
  • min tracking stack

Mental Model:
min_stack stores historical minimums.

def __init__(self):
    self.stack = []
    self.min_stack = []

def push(self, val: int) -> None:
    self.stack.append(val)

    if not self.min_stack or val <= self.min_stack[-1]:
        self.min_stack.append(val)

def pop(self) -> None:
    val = self.stack.pop()

    if val == self.min_stack[-1]:
        self.min_stack.pop()

def top(self) -> int:
    return self.stack[-1]

def getMin(self) -> int:
    return self.min_stack[-1]

Runtime:

  • push -> O(1)
  • pop -> O(1)
  • top -> O(1)
  • getMin -> O(1)

Union Find / Disjoin Set Union (DSU)

find -> nearly O(1)
union -> nearly O(1)

def __init__(self, n):
    self.parent = [i for i in range(n)]

def find(self, x):
    if self.parent[x] != x:
        self.parent[x] = self.find(self.parent[x])  # path compression

    return self.parent[x]

def union(self, a, b):
    pa = self.find(a)
    pb = self.find(b)

    if pa != pb:
        self.parent[pb] = pa

Time-Based Key Value Store

set -> O(1)
get -> O(log n)

HashMap + Binary Search

  • fast insertion
  • efficient historical lookup

Purpose:
Store values with timestamps and retrieve most recent valid values.

Core Structure:

store[key] = [(timestamp, value)]

Key Insight:
Timestamps are inserted in sorted order.

Lookup:
Find:

  • largest timedstamp <= target timestamp

Uses:

  • binary search
  • bisect_right
from collections import defaultdict
import bisect

class TimeMap:
  def __init__(self):
    self.store = defaultdict(list)

  def set(self, key: str, value: str, timestamp: int) -> None:
    self.store[key].append((timestamp, value))

  def get(self, key: str, timestamp: int) -> str:
    values = self.store[key]

    idx = bisect.bisect_right(
        values,
        (timestamp, chr(255))
    ) - 1

    if idx >= 0:
        return values[idx][1]

    return ""

Why bisect_right?

  • Need rightmost valid timestamp.

Mental Model:

  • Binary search over historical version

Common Signals:

  • historical lookup
  • versioned state
  • timestamp queries

collections.OrderedDict - maintains insertion order efficiently.
