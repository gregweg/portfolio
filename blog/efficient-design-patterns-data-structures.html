Core Idea

Maintain a stack in sorted/monotonic order while processing elements.
- Each element pushed and popped once, which usually gives O(n) instead of O(n^2)
- Monotonically Increasing Stack - values increase bottom -> top
- Monotonically Decreasing Stack - values decrease bottom -> top
- Most problems store INDICES, not values
- Allows access to nums[index]
- preserves positions
- needed for distance/span calculations

def finalPrices(self, prices: List[int]) -> List[int]:
    result = prices[:] # copy
    stack = []

    for i, price in enumerate(prices):
        # peak at top of stack
        while stack and prices[stack[-1]] >= price:
            idx = stack.pop()
            result[idx] -= price

        stack.append(i)
    return result

Template

parent = [i for i in range(n)]

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])  # path compression
    return parent[x]

def union(a, b):
    pa, pb = find(a), find(b)
    if pa != pb:
        parent[pb] = pa

# Greedy Scheduling
  # Sort by start time
  # - process sequentially
  # - maintain current active interval


  # Sort: O(n log n)
  # Scan: O(n)
  def merge(self, intervals: List[List[int]]) -> List[List[int]]:
    intervals.sort()
    merged = []

    for start, end in intervals:
      if not merged or start > merged[-1][1]:
        merged.append([start, end])
      else:
        merged[-1][1] = max(merged[-1][1], end)
    return merged

import random
class RandomizedSet:
  def __init__(self):
    self.indices = {}
    self.number_list = []
  # Runtime: O(1) average
  def insert(self, val: int) -> bool:
    if val in self.indices:
      return False
    self.indices[val] = len(self.number_list)
    self.number_list.append(val)
    return True
  # Runtime: O(1) average
  def remove(self, val: int) -> bool:
    if val not in self.indices:
      return False

    val_idx = self.indices[val]
    last_val = self.number_list[-1]

    self.number_list[val_idx] = last_val
    self.indices[last_val] = val_idx

    self.number_list.pop()
    del self.indices[val]

    return True
  # Runtime: O(1)
  def getRandom(self) -> int:
    return random.choice(self.number_list)

def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
  if not root or k == 0:
    return 0

  nodes = []

  def in_order(node: Optional[TreeNode]) -> None:
    if not node:
      return      

    in_order(node.left)
    nodes.append(node.val)
    if len(nodes) >= k:
      return
    in_order(node.right)

  in_order(root)
  return nodes[k-1]

        fill_island(r+dr, c+dc)

          if grid[r][c] == "1":
            island_count += 1
            fill_island(r,c)
    return island_count

# Word Search - DFS + Backtracking
#  Runtime: O(rows * cols * 4^len(word))
#  Space: O(len(word))
def exist(self, board: List[List[str]], word: str) -> bool:
  rows, cols = len(board), len(board[0])
  visited = set()

  def dfs(r, c, i):
    if i == len(word):
      return True

    if not(0 <= r < rows and 0 <= c < cols):
      return False

    if (r, c) in visited:
      return False

    if board[r][c] != word[i]:
      return False

    visited.add((r, c))

    found = (
      dfs(r + 1, c, i + 1) or
      dfs(r - 1, c, i + 1) or
      dfs(r, c + 1, i + 1) or
      dfs(r, c - 1, i + 1)
    )

    visited.remove((r, c))

    return found

  for r in range(rows):
    for c in range(cols):
      if dfs(r, c, 0):
        return True
  return False

def sortedSquares(self, nums: List[int]) -> List[int]:
  left, right = 0, len(nums) - 1
  res = [0] * len(nums)
  write = len(nums) - 1

  while left <= right:
    left_sq = nums[left] ** 2
    right_sq = nums[right] ** 2

    if left_sq > right_sq:
      res[write] = left_sq
      left += 1
    else:
      res[write] = right_sq
      right -= 1

    write -= 1

  return res
